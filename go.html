import time import random import pyautogui # 确保安装了此库：pip install pyautogui # --- 第一步：定义函数，告诉推销员具体的动作 --- def run_smart_mission(task_no): # 这里填入你屏幕上 1-5 号按钮的实际坐标 (x, y) # 你可以用 pyautogui.position() 提前测好位置 locations = { 1: (500, 400), 2: (500, 480), 3: (500, 560), 4: (500, 640), 5: (500, 720) } if task_no in locations: target_x, target_y = locations[task_no] # 模拟真人：先缓慢移动过去，再点击 pyautogui.moveTo(target_x, target_y, duration=random.uniform(0.5, 1.2)) pyautogui.click() print(f"✅ 已实际点击位置: {target_x}, {target_y}") # 点击后停留一会儿，模拟“加载页面”和“阅读” time.sleep(random.randint(5, 10)) # 如果需要点完后回到主页，可以加一个“返回”或“刷新”动作 # pyautogui.press('f5') # --- 第二步：主循环逻辑 --- if __name__ == "__main__": print("🚀 Web3Stroll CPA 模式启动...") while True: try: # 随机决定执行哪个任务 task_id = random.randint(1, 5) print(f"🎯 正在执行任务: {task_id}") # --- 调用上面定义的函数 --- run_smart_mission(task_id) # 冷却时间 wait_time = random.randint(25, 35) print(f"📅 任务结束，休眠 {wait_time} 秒...") time.sleep(wait_time) except Exception as e: print(f"❌ 运行异常: {e}") time.sleep(5)
